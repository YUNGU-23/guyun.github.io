class HitCounter:
    def __init__(self):
        self.time = [0] * 300
        self.count = [0] * 300

    def hit(self, timestamp: int):
        idx = timestamp % 300
        if self.time[idx] != timestamp:
            self.time[idx] = timestamp
            self.count[idx] = 1
        else:
            self.count[idx] += 1

    def get_hits(self, timestamp: int) -> int:
        res = 0
        for i in range(300):
            if self.time[i] > timestamp - 300:
                res += self.count[i]
        return res
