### LC 134. Gas Station
class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        # greedy approach, TC: O(n), SC: O(1)
        curSum, totalSum, res = 0, 0, 0
        for i in range(len(gas)):
            curSum += gas[i] - cost[i]
            totalSum += gas[i] - cost[i]
            if curSum < 0:
                res = i + 1
                curSum = 0
        if totalSum < 0: return -1
        return res
