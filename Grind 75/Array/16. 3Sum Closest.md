#### LC 16. 3Sum Closest
class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        # use three pointers, and sort nums
        # TC: O(n^2), SC: O(n)
        if len(nums) == 3: return sum(nums)
        nums.sort()
        res = float('inf')
        for i in range(len(nums) - 2):
            l, r = i + 1, len(nums) - 1
            while l < r: 
                total = nums[i] + nums[l] + nums[r]
                if total == target:
                    return total
                if abs(total - target) < abs(res - target):
                    res = total
                if total < target:
                    l += 1
                else:
                    r -= 1
        return res
