### LC 242. Valid Anagram
class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        # # using array, TC: O(n), SC: O(1)
        array = [0] * 26
        for i in s:
            array[ord('a') - ord(i)] += 1
        
        for i in t:
            array[ord('a') - ord(i)] -= 1
        
        for i in range(len(array)):
            if array[i] != 0:
                return False
        return True
