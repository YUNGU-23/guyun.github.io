### LC 3. Longest Substring Without Repeating Characters
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        # use set and sliding window, TC: O(n), SC: O(n)
        strSet = set()
        res = 0
        l = 0
        for r in range(len(s)):
            while s[r] in strSet:
                strSet.remove(s[l])
                l += 1
            strSet.add(s[r])
            res = max(res, r - l + 1)
        return res
