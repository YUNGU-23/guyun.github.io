### LC 215. Kth Largest Element in an Array
```
class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:
        # TC: O(n log n), SC: O(n)
        minHeap = []
        for x, y in points:
            dist = x ** 2 + y ** 2
            minHeap.append([dist, x, y])
        heapq.heapify(minHeap)

        res = []
        while k:
            dist, x, y = heapq.heappop(minHeap)
            res.append([x, y])
            k -= 1
        return res
```
