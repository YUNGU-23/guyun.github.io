### LC 51. N-Queens
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        # Use DFS, TC: O(n^2), SC: O(n^2)
        res = []
        col, posDiag, negDiag = set(), set(), set()
        board = [['.'] * n for _ in range(n)]

        def backTracking(r):
            if r == n:
                copy = [''.join(row) for row in board]
                res.append(copy)
                return
            
            for c in range(n):
                if c in col or (r + c) in posDiag or (r - c) in negDiag:
                    continue
                
                col.add(c)
                posDiag.add(r + c)
                negDiag.add(r - c)
                board[r][c] = 'Q'

                backTracking(r + 1)

                col.remove(c)
                posDiag.remove(r + c)
                negDiag.remove(r - c)
                board[r][c] = '.'

        backTracking(0)
        return res
