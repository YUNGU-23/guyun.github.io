class Solution:
    def minSteps(self, s: str, t: str) -> int:
        # TC: O(n), SC: O(1)
        count_s = [0] * 26
        count_t = [0] * 26
        for i in s:
            count_s[ord(i) - ord('a')] += 1
        for i in t:
            count_t[ord(i) - ord('a')] += 1
        res = 0
        for i in range(26):
            res += max(0, count_s[i] - count_t[i])
        return res
